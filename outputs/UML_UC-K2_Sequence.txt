```plantuml
@startuml
!pragma layout GAVIS

actor "Kiosk User" as User
participant "Kiosk UI" as UI
participant "Kiosk Backend" as Backend
participant "Game Service" as GameSvc
database "Kiosk DB" as KioskDB

title Game Selection and Configuration Flow (UC-K2)

User -> UI: Select game "<game_name>"

alt Basic Flow: Successful Game and Setting Selection
    UI -> Backend: Request game details (gameId)
    activate Backend
    Backend -> GameSvc: Get game details (gameId)
    activate GameSvc
    GameSvc --> Backend: Return game details (price, duration options, difficulty)
    deactivate GameSvc
    Backend --> UI: Return game details
    deactivate Backend
    UI -> User: Display game details and configuration options

    User -> UI: Select "<duration>" and "<difficulty>", then Confirm
    activate UI
    UI -> Backend: Save Game Configuration (gameId, duration, difficulty)
    activate Backend
    Backend -> KioskDB: Persist game configuration
    activate KioskDB
    KioskDB --> Backend: Configuration saved confirmation
    deactivate KioskDB
    Backend --> UI: Configuration saved acknowledgement
    deactivate Backend
    UI -> UI: Transition to "Zone Check" screen
    deactivate UI
    UI -> User: Display "Zone Check" screen

else Exception Flow I1: Backend Game Service Unreachable
    UI -> Backend: Request game details (gameId)
    activate Backend
    Backend -> GameSvc: Attempt to get game details (gameId)
    activate GameSvc
    GameSvc --[#red]> Backend: Service Unreachable Error
    deactivate GameSvc
    Backend --[#red]> UI: Error: Game Service Unreachable
    deactivate Backend
    UI -> UI: Retrieve "Last known details" from local cache
    UI -> User: Display cached details and allow proceeding with default/cached settings
end
@enduml
```

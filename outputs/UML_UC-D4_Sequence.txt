@startuml
!theme mars

actor "Manager/Admin" as User
participant "Frontend Application" as Frontend
participant "Backend Configuration Service" as Backend
participant "Validation Module" as Validation
database "Configuration Database" as DB

title Configuration Management (UC-D4) - Component Interaction Flow

' Scenario Outline: Updating various configuration settings
box "Scenario: Updating various configuration settings"
    activate Frontend
    User -> Frontend: Navigates to "<config_type>" settings page
    User -> Frontend: Selects item "<item_name>" and updates "<parameter>" to "<new_value>"
    User -> Frontend: Clicks "Save"

    Frontend -> Backend: POST /api/config/{config_type}/update (updatePayload)
    activate Backend
    Backend -> Validation: Validate(updatePayload)
    activate Validation
    Validation --> Backend: Validation successful (e.g., all fields valid, format correct)
    deactivate Validation

    Backend -> DB: Save updated configuration (updatePayload)
    activate DB
    DB --> Backend: Configuration saved successfully
    deactivate DB

    Backend --> Frontend: 200 OK (Update successful)
    deactivate Backend
    Frontend --> User: Displays "Saved successfully" confirmation
    deactivate Frontend
end box

' Scenario: Validation of campaign dates
alt Scenario: Validation of campaign dates (Invalid Date Range)
    activate Frontend
    User -> Frontend: Initiates creation of new pricing campaign
    User -> Frontend: Sets "End Date" before "Start Date"
    User -> Frontend: Clicks "Save" (or attempts to submit)

    Frontend -> Backend: POST /api/pricing/campaigns/create (newCampaignPayload)
    activate Backend
    Backend -> Validation: Validate(newCampaignPayload)
    activate Validation
    Validation --> Backend: Validation failed (Reason: Invalid Date Range)
    deactivate Validation

    Backend --> Frontend: 400 Bad Request (Error: Invalid Date Range)
    deactivate Backend
    Frontend --> User: Displays "Invalid Date Range" error
    Frontend --> User: Prevents saving the campaign (e.g., keeps form open, disables submit)
    deactivate Frontend
end alt

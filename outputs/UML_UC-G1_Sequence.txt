```plantuml
@startuml
title Game Session Initiation (UC-G1)

participant Kiosk
participant GameClient
participant "Backend Session Service" as Backend
participant "Game Engine" as GameEngine

' Background: Kiosk triggers command, GameClient is idle
activate Kiosk
Kiosk -> GameClient: "Start Session" Command
deactivate Kiosk

activate GameClient
GameClient -> GameClient: GameClient is in "Idle" state (initial state)

group Game Session Initiation Flow
    GameClient -> GameClient: Receive start command

    alt Successfully start game engine with backend parameters
        note right of GameClient: Scenario: Successfully start game engine
        GameClient -> Backend: Request game parameters (difficulty, duration)
        activate Backend
        Backend --> GameClient: Return valid parameters (difficulty, duration)
        deactivate Backend

        GameClient -> GameClient: Generate unique Session ID
        activate GameEngine
        GameClient -> GameEngine: Initialize and Start Gameplay (parameters, Session ID)
        GameEngine --> GameClient: Gameplay started successfully
        deactivate GameEngine
    else Start game using cached parameters when backend is down
        note right of GameClient: Scenario: Start game using cached parameters
        GameClient -> Backend: Request game parameters (difficulty, duration)
        activate Backend
        note right: Backend is unreachable or responds with error/timeout
        Backend -[dotted]> GameClient: Error / Timeout (Backend Unreachable)
        deactivate Backend

        GameClient -> GameClient: Load "Default" or "Last Known" parameters from local cache
        GameClient -> GameClient: Generate unique Session ID (for offline session)
        activate GameEngine
        GameClient -> GameEngine: Initialize and Start Gameplay (cached parameters, "Offline Mode")
        GameEngine --> GameClient: Gameplay started in "Offline Mode"
        deactivate GameEngine
    end
end

deactivate GameClient

@enduml
```

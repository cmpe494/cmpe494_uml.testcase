```plantuml
@startuml
!theme mars

title User Account Management - Component Interaction Flow

actor Admin as "Admin User"
actor Manager as "Manager User"

participant AdminUI as "Admin User Interface"
participant ManagerUI as "Manager User Interface"
participant UserService as "User Management Service"
participant AuthService as "Authorization Service"
participant PasswordService as "Password Service"
database Database as "User Database"

box "System Components"
    AdminUI -[hidden]-> UserService
    ManagerUI -[hidden]-> UserService
    UserService -[hidden]-> AuthService
    UserService -[hidden]-> PasswordService
    UserService -[hidden]-> Database
end box

group "Scenario: Successfully create a new staff account"
    Admin -> AdminUI: 1. Enters valid details for new "Cashier"
    Admin -> AdminUI: 2. Clicks "Save" button
    AdminUI -> UserService: 3. POST /staff-accounts (new_user_details)
    activate UserService
    UserService -> AuthService: 4. Verify Authorization (Admin, create:staff_account)
    activate AuthService
    AuthService --> UserService: 5. Authorization Granted
    deactivate AuthService
    UserService -> PasswordService: 6. Generate Temporary Password
    activate PasswordService
    PasswordService --> UserService: 7. Temporary Password (e.g., "tempXYZ")
    deactivate PasswordService
    UserService -> Database: 8. Save New User Account (user_details, "tempXYZ")
    activate Database
    Database --> UserService: 9. Account Created (user_id)
    deactivate Database
    UserService --> AdminUI: 10. 201 Created (User created successfully)
    deactivate UserService
    AdminUI -> Admin: 11. Display "User created successfully" confirmation
end

group "Scenario: Unauthorized account creation attempt"
    Manager -> ManagerUI: 1. Attempts to create account with role "Admin"
    ManagerUI -> UserService: 2. POST /staff-accounts (new_user_details, role=Admin)
    activate UserService
    UserService -> AuthService: 3. Verify Authorization (Manager, create:admin_account)
    activate AuthService
    AuthService --> UserService: 4. Authorization Denied (Insufficient privileges)
    deactivate AuthService
    UserService --> ManagerUI: 5. 403 Forbidden (Insufficient privileges)
    deactivate UserService
    ManagerUI -> Manager: 6. Display "Insufficient privileges" error message
end

group "Scenario: Resetting a user's password"
    Admin -> AdminUI: 1. Selects an existing user account (user_id)
    Admin -> AdminUI: 2. Clicks "Reset Password"
    AdminUI -> UserService: 3. POST /user/{user_id}/reset-password
    activate UserService
    UserService -> PasswordService: 4. Generate New Temporary Password
    activate PasswordService
    PasswordService --> UserService: 5. New Temporary Password (e.g., "newTempABC")
    deactivate PasswordService
    UserService -> Database: 6. Update User Password (user_id, "newTempABC", invalidate_old=true)
    activate Database
    Database --> UserService: 7. Password Updated Successfully
    deactivate Database
    UserService --> AdminUI: 8. 200 OK (Password reset confirmation)
    deactivate UserService
    AdminUI -> Admin: 9. Display "Password reset confirmation"
end

@enduml
```
